### 2. –†–∞—Å—à–∏—Ä–∏—Ç—å RedisService (`app/services/redis_service.py`)

–î–æ–±–∞–≤–∏—Ç—å –º–µ—Ç–æ–¥—ã –¥–ª—è —Ä–∞–±–æ—Ç—ã —Å –¥–∞–Ω–Ω—ã–º–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è:

```python
# –î–æ–±–∞–≤–∏—Ç—å –≤ –∫–ª–∞—Å—Å RedisService:

async def get_user_data(self, user_id: int) -> Optional[dict]:
    """–ü–æ–ª—É—á–∏—Ç—å –≤—Å–µ –¥–∞–Ω–Ω—ã–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –∏–∑ Redis"""

    return await self.get_json(key)

async def set_user_data(self, user_id: int, data: dict, expire: Optional[int] = None) -> bool:
    """–°–æ—Ö—Ä–∞–Ω–∏—Ç—å –¥–∞–Ω–Ω—ã–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –≤ Redis (–±–µ–∑ expire = –ø–æ—Å—Ç–æ—è–Ω–Ω–æ)"""
    key = f"user:{user_id}:data"
    return await self.set_json(key, data, expire=expire)

async def update_user_field(self, user_id: int, field: str, value: Any) -> bool:
    """–û–±–Ω–æ–≤–∏—Ç—å –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–µ –ø–æ–ª–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è"""
    user_data = await self.get_user_data(user_id) or {}
    user_data[field] = value
    return await self.set_user_data(user_id, user_data)

async def get_user_field(self, user_id: int, field: str, default: Any = None) -> Any:
    """–ü–æ–ª—É—á–∏—Ç—å –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–µ –ø–æ–ª–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è"""
    user_data = await self.get_user_data(user_id)
    if user_data:
        return user_data.get(field, default)
    return default

async def clear_user_data(self, user_id: int) -> bool:
    """–ü–æ–ª–Ω–∞—è –æ—á–∏—Å—Ç–∫–∞ –≤—Å–µ—Ö –¥–∞–Ω–Ω—ã—Ö –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è –∏–∑ Redis"""
    keys_to_delete = [
        f"user:{user_id}:data",
        f"user:{user_id}:generations",  # —Å—Ç–∞—Ä—ã–π –∫–ª—é—á, –µ—Å–ª–∏ –±—ã–ª
    ]
    
    deleted_count = 0
    for key in keys_to_delete:
        if await self.exists(key):
            await self.delete(key)
            deleted_count += 1
    
    logger.info(f"Cleared {deleted_count} Redis keys for user {user_id}")
    return deleted_count > 0
```

### 4. –û–±–Ω–æ–≤–∏—Ç—å –∫–æ–º–∞–Ω–¥—É `/clear` (`app/bot/handlers/commands.py`)

```python
    """–û–±—Ä–∞–±–æ—Ç—á–∏–∫ –∫–æ–º–∞–Ω–¥—ã /clear - –æ—á–∏—â–∞–µ—Ç —Ñ–æ—Ç–æ –∏–∑ –ë–î –∏ –¥–∞–Ω–Ω—ã–µ –∏–∑ Redis"""

    from app.models.user import User
    from app.models.photo import UserPhoto
    from app.services.redis_service import redis_service
    from sqlalchemy import select, delete
    
    user = message.from_user
    
    try:
        # –£–¥–∞–ª—è–µ–º —Ñ–æ—Ç–æ –∏–∑ –ë–î

                select(User).where(User.telegram_id == user.id)

            db_user = result.scalar_one_or_none()

                await session.execute(
                    delete(UserPhoto).where(UserPhoto.user_id == db_user.id)
                )
                await session.commit()

        # –û—á–∏—â–∞–µ–º –¥–∞–Ω–Ω—ã–µ –∏–∑ Redis

        # –û—á–∏—â–∞–µ–º FSM —Å–æ—Å—Ç–æ—è–Ω–∏–µ

        await message.answer(
            "üßπ <b>–î–∞–Ω–Ω—ã–µ –æ—á–∏—â–µ–Ω—ã!</b>\n\n–í—Å–µ –∑–∞–≥—Ä—É–∂–µ–Ω–Ω—ã–µ —Ñ–æ—Ç–æ, –∫–µ—à –∏ —Å–æ—Å—Ç–æ—è–Ω–∏—è —Å–±—Ä–æ—à–µ–Ω—ã.\n–ò—Å–ø–æ–ª—å–∑—É–π /start –¥–ª—è –Ω–∞—á–∞–ª–∞ —Ä–∞–±–æ—Ç—ã.",

        await message.answer(
            "‚ö†Ô∏è –ü—Ä–æ–∏–∑–æ—à–ª–∞ –æ—à–∏–±–∫–∞ –ø—Ä–∏ –æ—á–∏—Å—Ç–∫–µ. –ü–æ–ø—Ä–æ–±—É–π –µ—â–µ —Ä–∞–∑.",
```
